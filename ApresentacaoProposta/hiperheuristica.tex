\subsection{Hiper-heurística}
\frame{
	\frametitle{Hiper heurística}
	\begin{block}{}
	 `` ... um método de busca ou mecanismo de aprendizado para seleção ou 
geração de heurísticas para problemas de busca'' \cite{burke2013hyper} 
	\end{block}
}

%%===================CF==============%%
\frame{
    \frametitle{Técnicas Selecionadas}
    \begin{block}{Choise Function (CF)}
      \begin{equation*}
      f(h_i) = \alpha f_1(h_i) + \delta f_2(h_i)
      \end{equation*}
    
      \begin{enumerate}
       \item $f_1$: desempenho da heurística de baixo nível;
       \item $f_2$: tempo decorrido desde a última vez em que foi selecionada.
      \end{enumerate}
    \end{block}
    
    \begin{block}{Random (R)}
    	Aleatoriamente seleciona uma heurística de baixo nível.
    \end{block}
}
